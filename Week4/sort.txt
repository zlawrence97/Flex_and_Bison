p program sort (arr, i, j, temp) ;

var i, j, temp: integer;

var arr: array [0..4] of integer;

begin
arr = { 6, 7, 5, 4, 1 };
print(arr);
while i < 5 do { j = i; 
	while j > 0 do { 
		if arr[j] < arr[j-1] 
			then { temp = arr[j]; arr[j] = arr[j-1]; arr[j-1] = temp; }; 
		j = j-1;}; 
	i = i+1; };
print(arr)
end
